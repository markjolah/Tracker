230 LAPTrack::SpMatT
231 LAPTrack::computeF2FCostMat(IdxT curFrame, IdxT nextFrame) const
232 {
233  IdxT nCur = nFrameLocs(curFrame-firstFrame);
234  IdxT nNext = nFrameLocs(nextFrame-firstFrame);
235  IdxT nTot = nCur+nNext;
236  //These will be our sparse matrix format vectors
237  std::vector<arma::uword> row_index;
238  std::vector<arma::uword> col_index;
239  std::vector<FloatT> values;
240  IdxT reserve_size = nTot+2*std::min(nCur*nNext, std::max(nCur,nNext)*10); //Guesstimate amount of entries used
241  row_index.reserve(reserve_size);
242  col_index.reserve(reserve_size);
243  values.reserve(reserve_size);
244 
245  IdxT deltaT = nextFrame - curFrame; //The number of frames spanned in the link
246  FloatT DdT = 2*D*deltaT;
247  FloatT position_gaussian_exponent_cuttoff = (maxPositionDisplacementSigma*maxPositionDisplacementSigma)/2.; //Only allow connections within 5 sigma
248  VecT feature_gaussian_exponent_cuttoff = (maxFeatureDisplacementSigma%maxFeatureDisplacementSigma)/2.; //Only allow connections within 5 sigma
249  FloatT norm_const = (nDims+nFeatures)*log2pi; //Pre-compute this
250 
251  //Fill in connection costs
252  const IVecT &curFrameLocs = frameLocIdx(curFrame-firstFrame);
253  const IVecT &nextFrameLocs = frameLocIdx(nextFrame-firstFrame);
254 // std::cout<<"nCur:"<<nCur<<" nNext:"<<nNext<<"\n";
255 // std::cout<<"nFrameLocs:"<<nFrameLocs.t()<<"\n";
256  //connecting locI in current frame to locJ in next frame
257  for(IdxT j=0; j<nNext; j++) {
258  IdxT next_idx = nextFrameLocs(j);
259  for(IdxT i=0; i<nCur; i++){
260  IdxT cur_idx = curFrameLocs(i);
261 // std::cout<<"i:"<<i<<" j:"<<j<<" curIdx:"<<cur_idx<<" nextIdx:"<<next_idx<<" DdT:"<<DdT<<"\n";
262  FloatT C=0;
263  bool feasible = true;
264  FloatT total_dist_sq=0;
265  for(IdxT d=0; d<nDims; d++){
266  FloatT dist_var = DdT + SE_position(cur_idx,d) + SE_position(next_idx,d);
267  FloatT dist = position(cur_idx,d) - position(next_idx,d);
268  FloatT dist_sq = dist*dist;
269  total_dist_sq += dist_sq;
270  FloatT cost_exponent = dist_sq/dist_var;
271 // std::cout<<"Dim:"<<d<<" dist:"<<dist<<" dist_var:"<<dist_var<<" costExp:"<<cost_exponent<<" ExpCuttoff:"<<position_gaussian_exponent_cuttoff<<"\n";
272 // std::cout<<"SE_position(cur_idx,d):"<<SE_position(cur_idx,d)<<"SE_position(next_idx,d):"<<SE_position(next_idx,d)<<"\n";
273  if(cost_exponent > position_gaussian_exponent_cuttoff) { //Too far away to be connected
274  feasible=false;
275  break;
276  }
277  C+= cost_exponent + log(dist_var);
278 // std::cout<<"cost: "<<C<<"\n";
279  }
280  if(!feasible) continue; //gaussian sigma constraint violated: move to next pair.
281  if(maxSpeed>0 && sqrt(total_dist_sq)/deltaT > maxSpeed) continue; //maxSpeed constraint violated
282  for(IdxT f=0; f<nFeatures; f++){
283  FloatT feat_var = featureVar(f) + SE_feature(cur_idx,f)+ SE_feature(next_idx,f);
284  FloatT feat_dist = feature(cur_idx,f) - feature(next_idx,f);
285  FloatT cost_exponent = feat_dist*feat_dist/feat_var;
286  if(cost_exponent > feature_gaussian_exponent_cuttoff(f)) { //Too far away to be connected
287  feasible=false;
288  break;
289  }
290  C+= cost_exponent + log(feat_var);
291  }
292  if(!feasible) continue; //move to next pair.
293  //Otherwise we have a valid cost so normalize and record it
294  C+= norm_const;
295  C*= 0.5;
296  C-= log1mkoff;
297 // std::cout<<"C:"<<C<<" log1mkoff:"<<log1mkoff<<"\n";
298  //Record cost
299  row_index.push_back(i);
300  col_index.push_back(j);
301  values.push_back(C);
302  //Record lower right block dummy cost
303  row_index.push_back(nCur+j);
304  col_index.push_back(nNext+i);
305  values.push_back(cost_epsilon);
306  }
307  }
308  //Fill in death costs
309  FloatT deathC= -logkoff;
310  for(IdxT i=0; i<nCur; i++){
311  row_index.push_back(i);
312  col_index.push_back(nNext+i);
313  values.push_back(deathC);
314  }
315  //Fill in birth costs
316  FloatT birthC = -logrho-logkon;
317  for(IdxT j=0; j<nNext; j++){
318  row_index.push_back(nCur+j);
319  col_index.push_back(j);
320  values.push_back(birthC);
321  }
322  //Assemble sparse matrix
323  IdxT nnz = values.size();
324  UMatT locations(2,nnz);
325  for(IdxT n=0; n<nnz;n++){
326  locations(0,n) = row_index[n];
327  locations(1,n) = col_index[n];
328  }
329  VecT values_vec(values.data(), nnz, false);//Re-use the vector's memory directly.
330  bool sort_them = true; //Make sure armadillo sorts the locations
331  bool check_for_zeros = false; //Don't bother checking for zeros
332  return {locations, values_vec, static_cast<arma::uword>(nTot), static_cast<arma::uword>(nTot), sort_them, check_for_zeros};
333 }

396  tracks[m].splice(tracks[m].end(),tracks[n]);
397  nNewTracks--;

407  //These are all now invalid since tracks variable is changed
408  trackAssignment.clear();

459  FloatT dist = position(locI,d) - position(locJ,d);
460  FloatT dist_sq = dist*dist;
461  total_dist_sq += dist_sq;
462  FloatT cost_exponent = dist_sq/dist_var;
463 // std::cout<<"Dim:"<<d<<" dist:"<<dist<<" dist_var:"<<dist_var<<" costExp:"<<cost_exponent<<" ExpCuttoff:"<<gaussian_exponent_cuttoff<<"\n";
464 // std::cout<<"SE_position(cur_idx,d):"<<SE_position(cur_idx,d)<<"SE_position(next_idx,d):"<<SE_position(next_idx,d)<<"\n";
465  if(cost_exponent > position_gaussian_exponent_cuttoff) { //Too far away to be connected
466  feasible=false;
467  break;
468  }
469  C+= cost_exponent + log(dist_var);
470  }
471  if(!feasible) continue; //gaussian sigma constraint violated: move to next pair.
472  if(maxSpeed>0 && sqrt(total_dist_sq)/deltaT > maxSpeed) continue; //maxSpeed constraint violated
473  for(IdxT f=0; f<nFeatures; f++){
474  FloatT feat_var = featureVar(f) + SE_feature(locI,f)+ SE_feature(locJ,f);
475  FloatT feat_dist = feature(locI,f) - feature(locJ,f);
476  FloatT cost_exponent = feat_dist*feat_dist/feat_var;
477  if(cost_exponent > feature_gaussian_exponent_cuttoff(f)) { //Too far away to be connected
478  feasible=false;
479  break;
480  }
481  C+= cost_exponent + log(feat_var);
482  }
483  if(!feasible) continue; //move to next pair.
484  //Otherwise we have a valid cost so normalize and record it
485  C+= norm_const;
486  C*= 0.5;
487  C-= logkon +logkoff*deltaT;
488  //Record cost
489  row_index.push_back(i);
490  col_index.push_back(j);
491  values.push_back(C);
492  //Record lower right block dummy cost
493  row_index.push_back(nTracks+j);
494  col_index.push_back(nTracks+i);
495  values.push_back(cost_epsilon);
496  }
497  }
498 
499  for(IdxT i=0; i<nTracks; i++){
500  //Fill in death costs
501  row_index.push_back(i);
502  col_index.push_back(nTracks+i);
503  values.push_back(deathC);
504  //Fill in birth costs
505  row_index.push_back(nTracks+i);
506  col_index.push_back(i);
507  values.push_back(birthC);
508  }
509  IdxT nnz = values.size();
510  UMatT locations(2,nnz);
511  for(IdxT n=0; n<nnz;n++){
512  locations(0,n) = row_index[n];
513  locations(1,n) = col_index[n];
514  }
515  bool sort_them = true; //Make sure armadillo sorts the locations
516  bool check_for_zeros = false; //Don't bother checking for zeros
517  return {locations, VecT(values), 2*static_cast<arma::uword>(nTracks), 2*static_cast<arma::uword>(nTracks), sort_them, check_for_zeros};
